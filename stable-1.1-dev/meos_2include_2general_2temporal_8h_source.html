187/* Variation flags
188 * The first byte of the variation flag depends on the type we are sending
189 * - Set types: xxxO where O states whether the set is ordered or not
190 * - Box types: xxTX where X and T state whether the corresponding dimensions
191 * are present
192 * - Temporal types: xxSS where SS correspond to the subtype
193 * and x are unused bits
194 */
195#define MEOS_WKB_ORDERED 0x01 // 1
196#define MEOS_WKB_XFLAG 0x01 // 1
197#define MEOS_WKB_TFLAG 0x02 // 2
198#define MEOS_WKB_INTERPFLAGS 0x0C // 4 + 8
199#define MEOS_WKB_ZFLAG 0x10 // 16
200#define MEOS_WKB_GEODETICFLAG 0x20 // 32
201#define MEOS_WKB_SRIDFLAG 0x40 // 64
202
203#define MEOS_WKB_GET_INTERP(flags) (((flags) & MEOS_WKB_INTERPFLAGS) >> 2)
204#define MEOS_WKB_SET_INTERP(flags, value) ((flags) = (((flags) & ~MEOS_WKB_INTERPFLAGS) | ((value & 0x0003) << 2)))
