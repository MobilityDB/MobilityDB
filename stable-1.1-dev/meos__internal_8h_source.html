50/*
51 * Obtain a geometry/geography point from the GSERIALIZED WITHOUT creating
52 * the corresponding LWGEOM. These functions constitute a **SERIOUS**
53 * break of encapsulation but it is the only way to achieve reasonable
54 * performance when manipulating mobility data.
55 * The datum_* functions suppose that the GSERIALIZED has been already
56 * detoasted. This is typically the case when the datum is within a Temporal*
57 * that has been already detoasted with PG_GETARG_TEMPORAL*
58 * The first variant (e.g. datum_point2d) is slower than the second (e.g.
59 * datum_point2d_p) since the point is passed by value and thus the bytes
60 * are copied. The second version is declared const because you aren't allowed
61 * to modify the values, only read them.
62 */
63

69#define GserializedPGetDatum(X) PointerGetDatum(X)

93/*****************************************************************************
94 * Macros for manipulating the 'flags' element where the less significant
95 * bits are GTZXIICB, where
96 * G: coordinates are geodetic
97 * T: has T coordinate,
98 * Z: has Z coordinate
99 * X: has value or X coordinate
100 * II: interpolation, whose values are
101 * - 00: INTERP_NONE (undetermined) for TInstant
102 * - 01: DISCRETE
103 * - 10: STEP
104 * - 11: LINEAR
105 * C: continuous base type / Ordered set
106 * B: base type passed by value
107 * Notice that the interpolation flags are only needed for sequence and
108 * sequence set subtypes.
109 *****************************************************************************/
110
111/* The following flag is only used for Set and TInstant */
112#define MEOS_FLAG_BYVAL 0x0001 // 1
113/* The following flag is only used for Set */
114#define MEOS_FLAG_ORDERED 0x0002 // 2
115/* The following flag is only used for Temporal */
116#define MEOS_FLAG_CONTINUOUS 0x0002 // 2
117/* The following two interpolation flags are only used for TSequence and TSequenceSet */
118#define MEOS_FLAGS_INTERP 0x000C // 4 / 8
119/* The following two flags are used for both bounding boxes and temporal types */
120#define MEOS_FLAG_X 0x0010 // 16
121#define MEOS_FLAG_Z 0x0020 // 32
122#define MEOS_FLAG_T 0x0040 // 64
123#define MEOS_FLAG_GEODETIC 0x0080 // 128
124
125#define MEOS_FLAGS_GET_BYVAL(flags) ((bool) (((flags) & MEOS_FLAG_BYVAL)))
126#define MEOS_FLAGS_GET_ORDERED(flags) ((bool) (((flags) & MEOS_FLAG_ORDERED)>>1))
127#define MEOS_FLAGS_GET_CONTINUOUS(flags) ((bool) (((flags) & MEOS_FLAG_CONTINUOUS)>>1))
128#define MEOS_FLAGS_GET_X(flags) ((bool) (((flags) & MEOS_FLAG_X)>>4))
129#define MEOS_FLAGS_GET_Z(flags) ((bool) (((flags) & MEOS_FLAG_Z)>>5))
130#define MEOS_FLAGS_GET_T(flags) ((bool) (((flags) & MEOS_FLAG_T)>>6))
131#define MEOS_FLAGS_GET_GEODETIC(flags) ((bool) (((flags) & MEOS_FLAG_GEODETIC)>>7))
132
133#define MEOS_FLAGS_BYREF(flags) ((bool) (((flags) & ! MEOS_FLAG_BYVAL)))
134
135#define MEOS_FLAGS_SET_BYVAL(flags, value) \
136 ((flags) = (value) ? ((flags) | MEOS_FLAG_BYVAL) : ((flags) & ~MEOS_FLAG_BYVAL))
137#define MEOS_FLAGS_SET_ORDERED(flags, value) \
138 ((flags) = (value) ? ((flags) | MEOS_FLAG_ORDERED) : ((flags) & ~MEOS_FLAG_ORDERED))
139#define MEOS_FLAGS_SET_CONTINUOUS(flags, value) \
140 ((flags) = (value) ? ((flags) | MEOS_FLAG_CONTINUOUS) : ((flags) & ~MEOS_FLAG_CONTINUOUS))
141#define MEOS_FLAGS_SET_X(flags, value) \
142 ((flags) = (value) ? ((flags) | MEOS_FLAG_X) : ((flags) & ~MEOS_FLAG_X))
143#define MEOS_FLAGS_SET_Z(flags, value) \
144 ((flags) = (value) ? ((flags) | MEOS_FLAG_Z) : ((flags) & ~MEOS_FLAG_Z))
145#define MEOS_FLAGS_SET_T(flags, value) \
146 ((flags) = (value) ? ((flags) | MEOS_FLAG_T) : ((flags) & ~MEOS_FLAG_T))
147#define MEOS_FLAGS_SET_GEODETIC(flags, value) \
148 ((flags) = (value) ? ((flags) | MEOS_FLAG_GEODETIC) : ((flags) & ~MEOS_FLAG_GEODETIC))
149
150#define MEOS_FLAGS_GET_INTERP(flags) (((flags) & MEOS_FLAGS_INTERP) >> 2)
151#define MEOS_FLAGS_SET_INTERP(flags, value) ((flags) = (((flags) & ~MEOS_FLAGS_INTERP) | ((value & 0x0003) << 2)))
152
153#define MEOS_FLAGS_DISCRETE_INTERP(flags) ((bool) (MEOS_FLAGS_GET_INTERP((flags)) == DISCRETE))
154#define MEOS_FLAGS_STEP_INTERP(flags) ((bool) (MEOS_FLAGS_GET_INTERP((flags)) == STEP))
155#define MEOS_FLAGS_LINEAR_INTERP(flags) ((bool) (MEOS_FLAGS_GET_INTERP((flags)) == LINEAR))
156#define MEOS_FLAGS_STEP_LINEAR_INTERP(flags) \
157 ((bool) (MEOS_FLAGS_GET_INTERP((flags)) == STEP || MEOS_FLAGS_GET_INTERP((flags)) == LINEAR))
