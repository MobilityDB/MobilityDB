93static inline size_t
94raquet_pixtype_size(MeosPixType pixtype)
95{
96 switch (pixtype)
97 {
98 case MEOS_PT_UINT8: return 1;
99 case MEOS_PT_INT16: return 2;
100 case MEOS_PT_INT32: return 4;
101 case MEOS_PT_FLOAT32: return 4;
102 case MEOS_PT_FLOAT64: return 8;
103 default: return 0;
104 }
105}
106
110static inline size_t
111raquet_pixels_size(const Raquet *rq)
112{
113 return (size_t) rq->width * rq->height *
114 raquet_pixtype_size((MeosPixType) rq->pixtype);
115}
