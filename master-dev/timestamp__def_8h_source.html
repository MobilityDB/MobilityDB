18/*
19 * Timestamp represents absolute time.
20 *
21 * Interval represents delta time. Keep track of months (and years), days,
22 * and hours/minutes/seconds separately since the elapsed time spanned is
23 * unknown until instantiated relative to an absolute time.
24 *
25 * Note that Postgres uses "time interval" to mean a bounded interval,
26 * consisting of a beginning and ending time, not a time span - thomas 97/03/20
27 *
28 * Timestamps, as well as the h/m/s fields of intervals, are stored as
29 * int64 values with units of microseconds. (Once upon a time they were
30 * double values with units of seconds.)
31 *
32 * TimeOffset and fsec_t are convenience typedefs for temporary variables.
33 * Do not use fsec_t in values stored on-disk.
34 * Also, fsec_t is only meant for *fractional* seconds; beware of overflow
35 * if the value you need to store could be many seconds.
36 */
37
38typedef int64 Timestamp;
39typedef int32 fsec_t; /* fractional seconds (in microseconds) */

42typedef int64 TimeOffset;

150/*
151 * Julian date support.
152 *
153 * date2j() and j2date() nominally handle the Julian date range 0..INT_MAX,
154 * or 4714-11-24 BC to 5874898-06-03 AD. In practice, date2j() will work and
155 * give correct negative Julian dates for dates before 4714-11-24 BC as well.
156 * We rely on it to do so back to 4714-11-01 BC. Allowing at least one day's
157 * slop is necessary so that timestamp rotation doesn't produce dates that
158 * would be rejected on input. For example, '4714-11-24 00:00 GMT BC' is a
159 * legal timestamptz value, but in zones east of Greenwich it would print as
160 * sometime in the afternoon of 4714-11-23 BC; if we couldn't process such a
161 * date we'd have a dump/reload failure. So the idea is for IS_VALID_JULIAN
162 * to accept a slightly wider range of dates than we really support, and
163 * then we apply the exact checks in IS_VALID_DATE or IS_VALID_TIMESTAMP,
164 * after timezone rotation if any. To save a few cycles, we can make
165 * IS_VALID_JULIAN check only to the month boundary, since its exact cutoffs
166 * are not very critical in this scheme.
167 *
168 * It is correct that JULIAN_MINYEAR is -4713, not -4714; it is defined to
169 * allow easy comparison to tm_year values, in which we follow the convention
170 * that tm_year <= 0 represents abs(tm_year)+1 BC.
171 */
172
173#define JULIAN_MINYEAR (-4713)
174#define JULIAN_MINMONTH (11)
175#define JULIAN_MINDAY (24)
176#define JULIAN_MAXYEAR (5874898)
177#define JULIAN_MAXMONTH (6)
178#define JULIAN_MAXDAY (3)
179
180#define IS_VALID_JULIAN(y,m,d) \
181 (((y) > JULIAN_MINYEAR || \
182 ((y) == JULIAN_MINYEAR && ((m) >= JULIAN_MINMONTH))) && \
183 ((y) < JULIAN_MAXYEAR || \
184 ((y) == JULIAN_MAXYEAR && ((m) < JULIAN_MAXMONTH))))
185

190/*
191 * Range limits for dates and timestamps.
192 *
193 * We have traditionally allowed Julian day zero as a valid datetime value,
194 * so that is the lower bound for both dates and timestamps.
195 *
196 * The upper limit for dates is 5874897-12-31, which is a bit less than what
197 * the Julian-date code can allow. For timestamps, the upper limit is
198 * 294276-12-31. The int64 overflow limit would be a few days later; again,
199 * leaving some slop avoids worries about corner-case overflow, and provides
200 * a simpler user-visible definition.
201 */
202
203/* First allowed date, and first disallowed date, in Julian-date form */
204#define DATETIME_MIN_JULIAN (0)
205#define DATE_END_JULIAN (2147483494) /* == date2j(JULIAN_MAXYEAR, 1, 1) */
206#define TIMESTAMP_END_JULIAN (109203528) /* == date2j(294277, 1, 1) */

208/* Timestamp limits */
209#define MIN_TIMESTAMP INT64CONST(-211813488000000000)

214/* Range-check a date (given in Postgres, not Julian, numbering) */
215#define IS_VALID_DATE(d) \
216 ((DATETIME_MIN_JULIAN - POSTGRES_EPOCH_JDATE) <= (d) && \

219/* Range-check a timestamp */
220#define IS_VALID_TIMESTAMP(t) (MIN_TIMESTAMP <= (t) && (t) < END_TIMESTAMP)
