216/* Variation flags
217 * The first byte of the variation flag depends on the type we are sending
218 * - Set types: xxxO where O states whether the set is ordered or not
219 * - Box types: xxTX where X and T state whether the corresponding dimensions
220 * are present
221 * - Temporal types: xxSS where SS correspond to the subtype
222 * and x are unused bits
223 */
224#define MEOS_WKB_ORDERED 0x01 // 1
225#define MEOS_WKB_XFLAG 0x01 // 1
226#define MEOS_WKB_TFLAG 0x02 // 2
227#define MEOS_WKB_INTERPFLAGS 0x0C // 4 + 8
228#define MEOS_WKB_ZFLAG 0x10 // 16
229#define MEOS_WKB_GEODETICFLAG 0x20 // 32
230#define MEOS_WKB_SRIDFLAG 0x40 // 64
231
232#define MEOS_WKB_GET_INTERP(flags) (((flags) & MEOS_WKB_INTERPFLAGS) >> 2)
233#define MEOS_WKB_SET_INTERP(flags, value) ((flags) = (((flags) & ~MEOS_WKB_INTERPFLAGS) | ((value & 0x0003) << 2)))
