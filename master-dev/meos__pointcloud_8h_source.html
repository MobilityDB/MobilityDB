33/*****************************************************************************
34 * Type definitions
35 *
36 * Opaque structures — binary-compatible with pgpointcloud's SERIALIZED_POINT
37 * and SERIALIZED_PATCH varlena layouts (see pointcloud-pg/pgsql/pc_pgsql.h).
38 * At the MEOS layer they are handled as opaque byte blobs; dimension-level
39 * access requires the pgpointcloud schema (looked up by pcid from the
40 * pointcloud_formats PG catalog table) and is not exposed here.
41 *****************************************************************************/
42
43typedef struct Pcpoint Pcpoint;
44typedef struct Pcpatch Pcpatch;
45/* Forward decl for the parsed pgpointcloud schema; full layout is in
46 * libpc.a's @c pc_api.h. Schema-aware MEOS helpers take this opaque
47 * pointer; obtain one via @ref meos_pc_schema. */
48typedef struct PCSCHEMA PCSCHEMA;

63typedef struct
64{
65 Span period;
66 double xmin;
67 double ymin;
68 double zmin;
69 double xmax;

78/* TPCBox shares its leading layout with STBox so that a TPCBox can be
79 * projected to an STBox by dropping the trailing pcid + padding. The
80 * layout-compatibility cast (CREATE CAST tpcbox AS stbox) and any
81 * future tooling that converts between the two rely on this. */
82static_assert(offsetof(TPCBox, period) == offsetof(STBox, period),
83 "TPCBox and STBox must share the period offset");
84static_assert(offsetof(TPCBox, xmin) == offsetof(STBox, xmin),
85 "TPCBox and STBox must share the xmin offset");
86static_assert(offsetof(TPCBox, zmax) == offsetof(STBox, zmax),
87 "TPCBox and STBox must share the zmax offset");
88static_assert(offsetof(TPCBox, srid) == offsetof(STBox, srid),
89 "TPCBox and STBox must share the srid offset");
