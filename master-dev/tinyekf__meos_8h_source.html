24static void _mulmat(
25 const _float_t * a,
26 const _float_t * b,
27 _float_t * c,
28 const int arows,
29 const int acols,
30 const int bcols)
31{
32 for (int i=0; i<arows; ++i) {
33 for (int j=0; j<bcols; ++j) {
34 c[i*bcols+j] = 0;
35 for (int k=0; k<acols; ++k) {
36 c[i*bcols+j] += a[i*acols+k] * b[k*bcols+j];
37 }
38 }
39 }
40}

102static int _choldc1(_float_t * a, _float_t * p, const int n)
103{
104 for (int i = 0; i < n; i++) {
105 for (int j = i; j < n; j++) {
106 _float_t sum = a[i*n+j];
107 for (int k = i - 1; k >= 0; k--) {
108 sum -= a[i*n+k] * a[j*n+k];
109 }
110 if (i == j) {
111 if (sum <= 0) {
112 return 1; /* error */
113 }
114 p[i] = sqrt(sum);
115 }
116 else {
117 a[j*n+i] = sum / p[i];
118 }
119 }
120 }
121
122 return 0; // success:w
123}

126static int _choldcsl(const _float_t * A, _float_t * a, _float_t * p, const int n)
127{
128 for (int i = 0; i < n; i++) {
129 for (int j = 0; j < n; j++) {
130 a[i*n+j] = A[i*n+j];
131 }
132 }
133 if (_choldc1(a, p, n)) {
134 return 1;
135 }
136 for (int i = 0; i < n; i++) {
137 a[i*n+i] = 1 / p[i];
138 for (int j = i + 1; j < n; j++) {
139 _float_t sum = 0;
140 for (int k = i; k < j; k++) {
141 sum -= a[j*n+k] * a[k*n+i];
142 }
143 a[j*n+i] = sum / p[j];
144 }
145 }
146
147 return 0; // success
148}

151static int _cholsl(const _float_t * A, _float_t * a, _float_t * p, const int n)
152{
153 if (_choldcsl(A,a,p,n)) {
154 return 1;
155 }
156
157 for (int i = 0; i < n; i++) {
158 for (int j = i + 1; j < n; j++) {
159 a[i*n+j] = 0.0;
160 }
161 }
162 for (int i = 0; i < n; i++) {
163 a[i*n+i] *= a[i*n+i];
164 for (int k = i + 1; k < n; k++) {
165 a[i*n+i] += a[k*n+i] * a[k*n+i];
166 }
167 for (int j = i + 1; j < n; j++) {
168 for (int k = j; k < n; k++) {
169 a[i*n+j] += a[k*n+i] * a[k*n+j];
170 }
171 }
172 }
173 for (int i = 0; i < n; i++) {
174 for (int j = 0; j < i; j++) {
175 a[i*n+j] = a[j*n+i];
176 }
177 }
178
179 return 0; // success
180}

239static void ekf_predict(
248 ekf_t * ekf,
249 const _float_t fx[EKF_N],
250 const _float_t F[EKF_N*EKF_N],
251 const _float_t Q[EKF_N*EKF_N])
252{
253 // \hat{x}_k = f(\hat{x}_{k-1}, u_k)
254 memcpy(ekf->x, fx, EKF_N*sizeof(_float_t));
255
256 // P_k = F_{k-1} P_{k-1} F^T_{k-1} + Q_{k-1}
257
258 _float_t FP[EKF_N*EKF_N] = {};
259 _mulmat(F, ekf->P, FP, EKF_N, EKF_N, EKF_N);
260
261 _float_t Ft[EKF_N*EKF_N] = {};
262 _transpose(F, Ft, EKF_N, EKF_N);
263
264 _float_t FPFt[EKF_N*EKF_N] = {};
265 _mulmat(FP, Ft, FPFt, EKF_N, EKF_N, EKF_N);
266
267 _addmat(FPFt, Q, ekf->P, EKF_N, EKF_N);
268}

271static void ekf_update_step3(ekf_t * ekf, _float_t GH[EKF_N*EKF_N])
272{
273 _negate(GH, EKF_N, EKF_N);
274 _addeye(GH, EKF_N);
275 _float_t GHP[EKF_N*EKF_N];
276 _mulmat(GH, ekf->P, GHP, EKF_N, EKF_N, EKF_N);
277 memcpy(ekf->P, GHP, EKF_N*EKF_N*sizeof(_float_t));
278}

289static bool ekf_update(
290 ekf_t * ekf,
291 const _float_t z[EKF_M],
292 const _float_t hx[EKF_M],
293 const _float_t H[EKF_M*EKF_N],
294 const _float_t R[EKF_M*EKF_M])
295{
296 // G_k = P_k H^T_k (H_k P_k H^T_k + R)^{-1}
297 _float_t G[EKF_N*EKF_M];
298 _float_t Ht[EKF_N*EKF_M];
299 _transpose(H, Ht, EKF_M, EKF_N);
300 _float_t PHt[EKF_N*EKF_M];
301 _mulmat(ekf->P, Ht, PHt, EKF_N, EKF_N, EKF_M);
302 _float_t HP[EKF_M*EKF_N];
303 _mulmat(H, ekf->P, HP, EKF_M, EKF_N, EKF_N);
304 _float_t HpHt[EKF_M*EKF_M];
305 _mulmat(HP, Ht, HpHt, EKF_M, EKF_N, EKF_M);
306 _float_t HpHtR[EKF_M*EKF_M];
307 _addmat(HpHt, R, HpHtR, EKF_M, EKF_M);
308 _float_t HPHtRinv[EKF_M*EKF_M];
309 if (!invert(HpHtR, HPHtRinv)) {
310 return false;
311 }
312 _mulmat(PHt, HPHtRinv, G, EKF_N, EKF_M, EKF_M);
313
314 // \hat{x}_k = \hat{x_k} + G_k(z_k - h(\hat{x}_k))
315 _float_t z_hx[EKF_M];
316 _sub(z, hx, z_hx, EKF_M);
317 _float_t Gz_hx[EKF_M*EKF_N];
318 _mulvec(G, z_hx, Gz_hx, EKF_N, EKF_M);
319 _addvec(ekf->x, Gz_hx, ekf->x, EKF_N);
320
321 // P_k = (I - G_k H_k) P_k
322 _float_t GH[EKF_N*EKF_N];
323 _mulmat(G, H, GH, EKF_N, EKF_M, EKF_N);
324 ekf_update_step3(ekf, GH);
325
326 // success
327 return true;
328}
