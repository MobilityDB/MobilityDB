63/* GEOS-free analytic distance engine shared with the temporal circular
64 * buffer family: a moving disc (radius may be 0, i.e. a moving point)
65 * against a decomposed geometry, indexed by a Morton bucket hierarchy for
66 * the nearest-approach (knn) kernels and the generic R-tree elsewhere. */
67
73typedef struct
74{
75 double x1, y1, x2, y2;
76 double xmin, ymin, xmax, ymax;
77 bool is_poly;
78 /* Circular-arc edge (from a CIRCULARSTRING): when @p is_arc is true the edge
79 * is the arc of the circle centred at (@p acx, @p acy) with radius @p arad,
80 * from angle @p at0 to @p at1 traversed counterclockwise when @p accw. The
81 * endpoints (x1,y1)/(x2,y2) are the arc start/end; the bounding box already
82 * accounts for the arc bulge (cardinal extremes within the angular span). */
83 bool is_arc;
84 double acx, acy, arad, at0, at1;
85 bool accw;
86} GeoDistEdge;
