46/*****************************************************************************
47 * Public base JSON / JSONB / JSONPATH API
48 *
49 * The full base-type surface (json/jsonb input-output, the json accessors,
50 * jsonb conversions, and jsonpath query) is declared here -- in the public
51 * meos/include tree -- so that every binding generated from the MEOS-API
52 * catalog gets the complete API over the base jsonb type, not only the
53 * temporal tjsonb wrapper. The implementations live in the vendored pgtypes
54 * sources; the Jsonb/JsonPath/text/Numeric types come from <pgtypes.h> above.
55 *****************************************************************************/

184#if MEOS
185 #define VALIDATE_JSONBSET(set, ret) \
186 do { \
187 if (! ensure_not_null((void *) set) || \
188 ! ensure_set_isof_type((set), T_JSONBSET) ) \
189 return (ret); \
190 } while (0)
191#else
192 #define VALIDATE_JSONBSET(set, ret) \
193 do { \
194 assert(set); \
195 assert((set)->settype == T_JSONBSET); \
196 } while (0)
197#endif
198
205#if MEOS
206 #define VALIDATE_TJSONB(temp, ret) \
207 do { \
208 if (! ensure_not_null((void *) (temp)) || \
209 ! ensure_temporal_isof_type((Temporal *) (temp), T_TJSONB) ) \
210 return (ret); \
211 } while (0)
212#else
213 #define VALIDATE_TJSONB(temp, ret) \
214 do { \
215 assert(temp); \
216 assert(((Temporal *) (temp))->temptype == T_TJSONB); \
217 } while (0)
