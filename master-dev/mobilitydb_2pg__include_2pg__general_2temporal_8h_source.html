150/*
151 * MobilityDB reuses the typmod definitions from PostGIS using the two spare
152 * bits left for storing the temporal subtype.
153 *
154 * The 'typmod' is an int32_t used as follows:
155 * Plus/minus = Top bit.
156 * Spare bits = Next 2 bits -> Used in MobilityDB to store the temporal subtype
157 * SRID = Next 21 bits.
158 * TYPE = Next 6 bits.
159 * ZM Flags = Bottom 2 bits.
160 *
161#define TYPMOD_GET_SRID(typmod) ((((typmod) & 0x0FFFFF00) - ((typmod) & 0x10000000)) >> 8)
162#define TYPMOD_SET_SRID(typmod, srid) ((typmod) = (((typmod) & 0xE00000FF) | ((srid & 0x001FFFFF)<<8)))
163#define TYPMOD_GET_TYPE(typmod) ((typmod & 0x000000FC)>>2)
164#define TYPMOD_SET_TYPE(typmod, type) ((typmod) = (typmod & 0xFFFFFF03) | ((type & 0x0000003F)<<2))
165#define TYPMOD_GET_Z(typmod) ((typmod & 0x00000002)>>1)
166#define TYPMOD_SET_Z(typmod) ((typmod) = typmod | 0x00000002)
167#define TYPMOD_GET_M(typmod) (typmod & 0x00000001)
168#define TYPMOD_SET_M(typmod) ((typmod) = typmod | 0x00000001)
169#define TYPMOD_GET_NDIMS(typmod) (2+TYPMOD_GET_Z(typmod)+TYPMOD_GET_M(typmod))
170 *
171 */
172
173/* Get/set the temporal subtype from the 2 spare bits from PostGIS */
174#define TYPMOD_GET_TEMPSUBTYPE(typmod) ((typmod & 0x60000000)>>29)
175#define TYPMOD_SET_TEMPSUBTYPE(typmod, tempsubtype) ((typmod) = (((typmod) & 0x9FFFFFFF) | ((tempsubtype & 0x00000003)<<29)))
