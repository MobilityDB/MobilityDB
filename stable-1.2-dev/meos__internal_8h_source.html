111/*****************************************************************************
112 * Macros for manipulating the 'flags' element where the less significant
113 * bits are GTZXIICB, where
114 * G: coordinates are geodetic
115 * T: has T coordinate,
116 * Z: has Z coordinate
117 * X: has value or X coordinate
118 * II: interpolation, whose values are
119 * - 00: INTERP_NONE (undetermined) for TInstant
120 * - 01: DISCRETE
121 * - 10: STEP
122 * - 11: LINEAR
123 * C: continuous base type / Ordered set
124 * B: base type passed by value
125 * Notice that the interpolation flags are only needed for sequence and
126 * sequence set subtypes.
127 *****************************************************************************/
128
129/* The following flag is only used for Set and TInstant */
130#define MEOS_FLAG_BYVAL 0x0001 // 1
131/* The following flag is only used for Set */
132#define MEOS_FLAG_ORDERED 0x0002 // 2
133/* The following flag is only used for Temporal */
134#define MEOS_FLAG_CONTINUOUS 0x0002 // 2
135/* The following two interpolation flags are only used for TSequence and TSequenceSet */
136#define MEOS_FLAGS_INTERP 0x000C // 4 / 8
137/* The following two flags are used for both bounding boxes and temporal types */
138#define MEOS_FLAG_X 0x0010 // 16
139#define MEOS_FLAG_Z 0x0020 // 32
140#define MEOS_FLAG_T 0x0040 // 64
141#define MEOS_FLAG_GEODETIC 0x0080 // 128
142
143#define MEOS_FLAGS_GET_BYVAL(flags) ((bool) (((flags) & MEOS_FLAG_BYVAL)))
144#define MEOS_FLAGS_GET_ORDERED(flags) ((bool) (((flags) & MEOS_FLAG_ORDERED)>>1))
145#define MEOS_FLAGS_GET_CONTINUOUS(flags) ((bool) (((flags) & MEOS_FLAG_CONTINUOUS)>>1))
146#define MEOS_FLAGS_GET_X(flags) ((bool) (((flags) & MEOS_FLAG_X)>>4))
147#define MEOS_FLAGS_GET_Z(flags) ((bool) (((flags) & MEOS_FLAG_Z)>>5))
148#define MEOS_FLAGS_GET_T(flags) ((bool) (((flags) & MEOS_FLAG_T)>>6))
149#define MEOS_FLAGS_GET_GEODETIC(flags) ((bool) (((flags) & MEOS_FLAG_GEODETIC)>>7))
150
151#define MEOS_FLAGS_BYREF(flags) ((bool) (((flags) & ! MEOS_FLAG_BYVAL)))
152
153#define MEOS_FLAGS_SET_BYVAL(flags, value) \
154 ((flags) = (value) ? ((flags) | MEOS_FLAG_BYVAL) : ((flags) & ~MEOS_FLAG_BYVAL))
155#define MEOS_FLAGS_SET_ORDERED(flags, value) \
156 ((flags) = (value) ? ((flags) | MEOS_FLAG_ORDERED) : ((flags) & ~MEOS_FLAG_ORDERED))
157#define MEOS_FLAGS_SET_CONTINUOUS(flags, value) \
158 ((flags) = (value) ? ((flags) | MEOS_FLAG_CONTINUOUS) : ((flags) & ~MEOS_FLAG_CONTINUOUS))
159#define MEOS_FLAGS_SET_X(flags, value) \
160 ((flags) = (value) ? ((flags) | MEOS_FLAG_X) : ((flags) & ~MEOS_FLAG_X))
161#define MEOS_FLAGS_SET_Z(flags, value) \
162 ((flags) = (value) ? ((flags) | MEOS_FLAG_Z) : ((flags) & ~MEOS_FLAG_Z))
163#define MEOS_FLAGS_SET_T(flags, value) \
164 ((flags) = (value) ? ((flags) | MEOS_FLAG_T) : ((flags) & ~MEOS_FLAG_T))
165#define MEOS_FLAGS_SET_GEODETIC(flags, value) \
166 ((flags) = (value) ? ((flags) | MEOS_FLAG_GEODETIC) : ((flags) & ~MEOS_FLAG_GEODETIC))
167
168#define MEOS_FLAGS_GET_INTERP(flags) (((flags) & MEOS_FLAGS_INTERP) >> 2)
169#define MEOS_FLAGS_SET_INTERP(flags, value) ((flags) = (((flags) & ~MEOS_FLAGS_INTERP) | ((value & 0x0003) << 2)))
170
171#define MEOS_FLAGS_DISCRETE_INTERP(flags) ((bool) (MEOS_FLAGS_GET_INTERP((flags)) == DISCRETE))
172#define MEOS_FLAGS_STEP_INTERP(flags) ((bool) (MEOS_FLAGS_GET_INTERP((flags)) == STEP))
173#define MEOS_FLAGS_LINEAR_INTERP(flags) ((bool) (MEOS_FLAGS_GET_INTERP((flags)) == LINEAR))
174#define MEOS_FLAGS_STEP_LINEAR_INTERP(flags) \
175 ((bool) (MEOS_FLAGS_GET_INTERP((flags)) == STEP || MEOS_FLAGS_GET_INTERP((flags)) == LINEAR))
